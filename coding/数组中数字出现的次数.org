class Solution:
    def singleNumbers(self, nums: List[int]) -> List[int]:
        k = 0
        for n in nums:
            k ^= n
        mask = 1
        while (k & mask==0):
            mask <<= 1
        a, b = 0, 0
        for n in nums:
            if n & mask == 0:
                a ^= n
            else:
                b ^= n
        return [a, b]

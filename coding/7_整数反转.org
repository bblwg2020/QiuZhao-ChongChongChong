class Solution:
    def reverse(self, x: int) -> int:
        x = str(x)
        if x[0]=='-':
            x = x[0] + x[1:][::-1]
        else:
            x = x[::-1]
        try:
            x = int(x)
            if x >= -2**31 and x<=2**31-1:
                return x
            else:
                return 0
        except:
            return 0

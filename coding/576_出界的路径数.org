class Solution:
    def findPaths(self, m: int, n: int, N: int, i: int, j: int) -> int:
        def helper(memo, i, j, k):
            if (i, j, k) not in memo:
                if i < 0 or i > m-1 or j < 0 or j > n-1:
                    memo[(i, j, k)] = 1
                elif k > 0:
                    # move one step
                    for direc in self.directions:
                        memo[(i, j, k)] += helper(memo, i+direc[0], j+direc[1], k-1)
